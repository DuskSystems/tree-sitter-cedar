permit(
  principal,
  action == AlbumOp::"addPhoto",
  resource in Account::"jane"
)
when {
  (context.photo.filetype in ["JPEG", "PNG"] &&
   context.photo.filesize_mb <= 1)
  ||
  (context.photo.filetype == "RAW" &&
   context.photo.filesize_mb <= 100 &&
   principal in Group::"AVTeam")
};
