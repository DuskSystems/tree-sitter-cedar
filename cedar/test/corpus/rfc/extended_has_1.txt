permit(
  principal is User,
  action == Action::"preview",
  resource == Movie::"Blockbuster"
) when {
  principal has contactInfo &&
  principal.contactInfo has address &&
  principal.contactInfo.address has zip &&
  principal.contactInfo.address.zip == "90210"
};
